#include<stdio.h>
int main()
{
 int n,pos,element;
 int arr[100];
 scanf("%d",&n);
 for(int i=0;i<n;i++)
{
    scanf("%d",&arr[i]);
} 
    scanf("%d",&element);
    scanf("%d",&pos);
    for(int i=n;i>=pos;i--)
    {
        arr[i]=arr[i-1];
    }
    arr[pos-1]=element;
    n++;
    
    for(int i=0;i<n;i++)
    {
        printf("%d",arr[i]);
    }
 return 0;
 
}
